CREATE TABLE Emp (
    emp_id INT PRIMARY KEY,
    ename VARCHAR(100),
    street VARCHAR(100),
    city VARCHAR(100)
);

CREATE TABLE Works (
    emp_id INT,
    c_id INT,
    ename VARCHAR(100),
    cname VARCHAR(100),
    sal DECIMAL(10, 2),
    FOREIGN KEY (emp_id) REFERENCES Emp(emp_id)
);

CREATE TABLE Company (
    c_id INT PRIMARY KEY,
    cname VARCHAR(100),
    city VARCHAR(100)
);

CREATE TABLE Manager (
    mgr_id INT PRIMARY KEY,
    mgrname VARCHAR(100)
);
